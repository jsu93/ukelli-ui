## 父组件控制 Tabs

<Playground>
  {
    () => {

      return (
        <State initial={{
          activeTabIdx: 1
        }}>
          {
            ({ state, setState }) => {
              const { activeTabIdx } = state;
              return (
                <div>
                  <div>
                    <span className="btn theme" onClick={e => setState({
                      activeTabIdx: 2
                    })}>切换到最后一个 Tab</span>
                  </div>
                  <Tabs
                    activeTabIdx={activeTabIdx}
                    onChangeTab={idx => {
                      setState({
                        activeTabIdx: idx
                      })
                    }}>
                    <Tab label="切换到内容1">
                      内容1
                    </Tab>
                    <Tab label="切换到内容2">
                      内容2
                    </Tab>
                    <Tab label="切换到内容3">
                      内容3
                    </Tab>
                  </Tabs>
                </div>
              )
            }
          }
        </State>
      )
    }
  }
</Playground>
