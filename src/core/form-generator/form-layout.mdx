## 表单联动

<Playground>
  <State initial={{
    activeVal: '',
    formOptions: [
      {
        refs: ['startDate', 'endDate'],
        type: 'datetimeRange',
        title: '日期',
        required: true,
        range: ['2018-10-10', '2018-10-11']
      },
      {
        ref: 'ref1',
        type: 'radio',
        title: '单选控件',
        values: {
          value1: 'value1',
          value2: 'value2',
          value3: 'value3',
        }
      },
      {
        ref: 'ref2',
        type: 'select',
        title: '选择控件',
        values: {
          value1: 'value1',
          value2: 'value2',
          value3: 'value3',
        }
      },
      {
        ref: 'ref3',
        type: 'input',
        title: '输入',
        required: true,
        inputType: 'number'
      },
    ],
    formOptionsMapper: {
      value1: [
        {
          ref: 'test',
          type: 'input',
          title: '测试1'
        }
      ],
      value2: [
        {
          ref: 'test',
          type: 'input',
          title: '测试2'
        }
      ],
      value3: [
        {
          ref: 'test',
          type: 'input',
          title: '测试3'
        }
      ],
    }
  }}>
    {
      ({ state, setState }) => (
        <div>
          <h4>初始表单</h4>
          <FormLayout
            btnConfig={[
              {
                action: (formRef, actingRef) => {
                  console.log(formRef, actingRef);
                },
                text: '测试按钮',
                actingRef: 'forTest'
              }
            ]}
            onChange={(value, ref) => {
              if(ref == 'ref1' || ref == 'ref2') setState({
                activeVal: value[ref]
              })
            }}
            formOptions={state.formOptions}/>
          <hr/>
          {
            state.activeVal ? (
              <React.Fragment>
                <h4>联动表单</h4>
                <FormLayout
                  key={state.activeVal}
                  btnConfig={[
                    {
                      action: (formRef, actingRef) => {
                        console.log(formRef, actingRef);
                      },
                      text: '测试按钮',
                      actingRef: 'forTest'
                    }
                  ]}
                  formOptions={state.formOptionsMapper[state.activeVal]}/>
              </React.Fragment>
            ) : null
          }
        </div>
      )
    }
  </State>
</Playground>
