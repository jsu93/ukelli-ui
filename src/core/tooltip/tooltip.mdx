import { Playground, PropsTable } from 'docz'
import { State, Toggle } from 'react-powerplug'
import { ToolTip } from './';

# ToolTip

<PropsTable of={ToolTip} />

<Playground>
  {
    () => {
      return (
        <State initial={{
          isOpen: 'isOpen',
          position: 'left'
        }}>
          {
            ({ setState, state }) => {
              const { isOpen, position } = state;
              return (
                <div>
                  <span className="ms10 link" onClick={e => setState({
                    position: 'left'
                  })}>左边</span>
                  <span className="ms10 link" onClick={e => setState({
                    position: 'right'
                  })}>右边</span>
                  <span className="ms10 link" onClick={e => setState({
                    position: 'top'
                  })}>上</span>
                  <span className="ms10 link" onClick={e => setState({
                    position: 'bottom'
                  })}>下</span>
                  <hr />
                  <ToolTip
                    onClick={e => setState({
                      isOpen: !isOpen
                    })}
                    position={position}
                    title={!isOpen ? "打开书" : "关闭书"}
                    n={isOpen ? "book-open" : "book"}>
                    打开位置 {position}
                  </ToolTip>
                </div>
              )
            }
          }
        </State>
      )
    }
  }
</Playground>
